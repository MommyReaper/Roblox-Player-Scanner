-- Advanced information gathering
local function getAdvancedPlayerInfo(targetPlayer)
    local info = {
        Username = targetPlayer.Name,
        DisplayName = targetPlayer.DisplayName,
        UserId = targetPlayer.UserId,
        AccountAge = targetPlayer.AccountAge,
        MembershipType = targetPlayer.MembershipType.Name,
        Team = targetPlayer.Team and targetPlayer.Team.Name or "No Team",
        TeamColor = targetPlayer.Team and targetPlayer.Team.TeamColor.Color or Color3.new(1, 1, 1),
        Leaderstats = {},
        CharacterInfo = {},
        Position = "N/A",
        Health = "N/A",
        MaxHealth = "N/A",
        WalkSpeed = "N/A",
        JumpPower = "N/A",
        Gear = "N/A",
        BodyParts = {},
        Accessories = {},
        StatusEffects = "N/A",
        VoiceState = "Unknown",
        CharacterAppearanceId = targetPlayer.CharacterAppearanceId,
        FriendStatus = "Unknown",
        IsFriend = false,
        IsFollowing = false,
        FollowersCount = 0,
        FollowingCount = 0,
        PreviousTeams = {},
        ChatCount = 0,
        LastSeen = "Unknown"
    }
    
    -- Character-based info
    if targetPlayer.Character then
        -- Position
        if targetPlayer.Character:FindFirstChild("HumanoidRootPart") then
            local pos = targetPlayer.Character.HumanoidRootPart.Position
            info.Position = math.floor(pos.X) .. ", " .. math.floor(pos.Y) .. ", " .. math.floor(pos.Z)
        end
        
        -- Health
        local humanoid = targetPlayer.Character:FindFirstChildOfClass("Humanoid")
        if humanoid then
            info.Health = math.floor(humanoid.Health)
            info.MaxHealth = math.floor(humanoid.MaxHealth)
            info.WalkSpeed = humanoid.WalkSpeed
            info.JumpPower = humanoid.JumpPower
            info.StatusEffects = humanoid:GetAttribute("Status") or "None"
        end
        
        -- Body parts and accessories
        for _, part in pairs(targetPlayer.Character:GetChildren()) do
            if part:IsA("BodyPart") then
                table.insert(info.BodyParts, part.Name)
            elseif part:IsA("Accessory") then
                table.insert(info.Accessories, part.Name)
            end
        end
    end
    
    -- Leaderstats
    local leaderstats = targetPlayer:FindFirstChild("leaderstats")
    if leaderstats then
        for _, stat in pairs(leaderstats:GetChildren()) do
            if stat:IsA("IntValue") or stat:IsA("NumberValue") then
                info.Leaderstats[stat.Name] = stat.Value
            end
        end
    end
    
    -- Backpack/Gear
    local backpack = targetPlayer:FindFirstChild("Backpack")
    if backpack then
        local gearList = {}
        for _, item in pairs(backpack:GetChildren()) do
            if item:IsA("Tool") then
                table.insert(gearList, item.Name)
            end
        end
        if #gearList > 0 then
            info.Gear = table.concat(gearList, ", ")
        end
    end
    
    -- Additional user info (if available through API)
    spawn(function()
        pcall(function()
            local success, result = pcall(function()
                return game:GetService("Players"):GetFriendStatus(targetPlayer.UserId)
            end)
            if success then
                info.FriendStatus = result
                info.IsFriend = (result == "Friend")
            end
        end)
    end)
    
    return info
end

-- Export data function
local function exportPlayerData()
    local exportData = {
        ServerId = game.JobId,
        Timestamp = os.date("%Y-%m-%d %H:%M:%S"),
        Players = {}
    }
    
    for _, targetPlayer in pairs(Players:GetPlayers()) do
        local info = getAdvancedPlayerInfo(targetPlayer)
        table.insert(exportData.Players, {
            Username = info.Username,
            DisplayName = info.DisplayName,
            UserId = info.UserId,
            AccountAge = info.AccountAge,
            MembershipType = info.MembershipType,
            Team = info.Team,
            Position = info.Position,
            Health = info.Health,
            Items = info.Gear,
            Stats = info.Leaderstats
        })
    end
    
    StarterGui:SetCore("ChatMakeSystemMessage", {
        Text = "[Scanner] Data exported to console (check F9)",
        Color = Color3.fromRGB(100, 255, 100)
    })
    
    print("--- PLAYER DATA EXPORT ---")
    print(HttpService:JSONEncode(exportData))
    print("--- END EXPORT ---")
end
